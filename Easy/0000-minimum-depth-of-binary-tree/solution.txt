 *     }
 * }
 */
class Solution {
    public int minDepth(TreeNode root) {
         if(root==null)
         return 0;
         int dp=minDepth(root.left);
         int gp=minDepth(root.right);
         if(root.left==null){
            return gp+1;
         }
         return Math.min(dp,gp)+1;
        
    }
}
