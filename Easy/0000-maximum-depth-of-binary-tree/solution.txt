 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int maxDepth(TreeNode root) {
        if(root==null){
            return 0;
        }
        int ld=maxDepth(root.left);
        int gd=maxDepth(root.right);
        return Math.max(ld,gd)+1;
