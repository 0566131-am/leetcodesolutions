#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def searchBST(self, root: Optional[TreeNode], val: int) -> Optional
    [TreeNode]:
        curr = root
        
        while curr:
            if curr.val == val:
                return curr  # Found the node
            elif val < curr.val:
                curr = curr.left  # Move left
            else:
                curr = curr.right  # Move right
                
        return None  # Value doesn't exist in the BST
