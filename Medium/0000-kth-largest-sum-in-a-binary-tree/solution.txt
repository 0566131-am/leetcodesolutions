                currentLevelSum += curr.val;
                
                if (curr.left != null) queue.offer(curr.left);
                if (curr.right != null) queue.offer(curr.right);
            }
            
            levelSums.add(currentLevelSum);
        } 
        if (levelSums.size() < k) {
            return -1;
        }
        Collections.sort(levelSums);
        
    }
        return levelSums.get(levelSums.size() - k);
}
