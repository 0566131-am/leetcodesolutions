            curr = nextNode;
        }
        
        // 3. Compare pairs from the first half and reversed second half
        int maxTwinSum = 0;
        ListNode firstHalf = head;
        ListNode secondHalf = prev; // 'prev' is now the head of the 
        reversed half
        
        while (secondHalf != null) {
            maxTwinSum = Math.max(maxTwinSum, firstHalf.val + secondHalf.
            val);
            firstHalf = firstHalf.next;
            secondHalf = secondHalf.next;
        }
        
        return maxTwinSum;
    }
}
