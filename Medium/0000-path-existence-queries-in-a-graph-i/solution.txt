            comp[i] = componentId;
        }
        
        // Process each query in O(1) time
        boolean[] answer = new boolean[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int u = queries[i][0];
            int v = queries[i][1];
            answer[i] = (comp[u] == comp[v]);
        }
        
        return answer;
    }
}
