            dist1 = Math.min(dist1, n - dist1);

            // check next
            int next = list.get((idx + 1) % size);
            int dist2 = Math.abs(q - next);
            dist2 = Math.min(dist2, n - dist2);

            minDist = Math.min(dist1, dist2);

            res.add(minDist);
        }

        return res;
    }
}
