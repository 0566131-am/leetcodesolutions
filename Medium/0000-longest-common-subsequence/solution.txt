        }

        if (text1.charAt(i) == text2.charAt(j)) {
            memo[i][j] = 1 + backtracking(text1, text2, i + 1, j + 1);
        } else {
            memo[i][j] = Math.max(
                backtracking(text1, text2, i + 1, j), 
                backtracking(text1, text2, i, j + 1)
            );
        }

        return memo[i][j];
    }
}
