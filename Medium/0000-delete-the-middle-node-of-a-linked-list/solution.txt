 */
class Solution {
    public ListNode deleteMiddle(ListNode head) {
        ListNode temp=new ListNode(0);
        temp.next=head;
        ListNode fast=head;
        ListNode slow=temp;
        while(fast!=null && fast.next!=null){
            slow =slow.next;
        }
            fast=fast.next.next;
        slow.next=slow.next.next;
        return temp.next;
    }
}
