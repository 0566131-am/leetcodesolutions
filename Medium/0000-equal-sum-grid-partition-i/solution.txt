
        // Check vertical cuts
        prefix = 0;
        for (int j = 0; j < n; j++) {
            for (int i = 0; i < m; i++) {
                prefix += grid[i][j];
            }
            if (prefix * 2 == total && j < n - 1) {
                return true;
            }
        }

        return false;
    }
}
