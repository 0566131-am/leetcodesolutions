        return result;
    }
    private boolean canShip(int[] weights, int days, int capacity) {
        int daysNeeded = 1;
        int currentLoad = 0;
        
        for (int w : weights) {
            if (currentLoad + w > capacity) {
                daysNeeded++;        
                currentLoad = 0;   
            }
            currentLoad += w;
        }
        return daysNeeded <= days;
    }
