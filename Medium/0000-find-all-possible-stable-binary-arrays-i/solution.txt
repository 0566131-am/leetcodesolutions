            if (z > 0 && len < limit)
                ans += dfs(z - 1, o, 0, len + 1);

            if (o > 0)
                ans += dfs(z, o - 1, 1, 1);
        } else {
            if (o > 0 && len < limit)
                ans += dfs(z, o - 1, 1, len + 1);

            if (z > 0)
                ans += dfs(z - 1, o, 0, 1);
        }

        return memo[z][o][last][len] = (int)(ans % MOD);
    }
}
