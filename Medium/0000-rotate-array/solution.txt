        private void reverse(int nums[],int left,int right){
            while(left<right){
                int temp=nums[left];
                nums[left]=nums[right];
                nums[right]=temp;
                left++;
                right--;
    
    }        
        reverse(nums,0,k-1);
        reverse(nums,k,n-1);
            }
        
        
    }
