                    board[r][c] = 'O'; 
                }
            }
        }
    }
    
    private void dfs(char[][] board, int r, int c) {
        if (r < 0 || r >= board.length || c < 0 || c >= board[0].length || board[r][c] != 'O') {
            return;
        }
        board[r][c] = 'E';
        dfs(board, r - 1, c);
        dfs(board, r + 1, c); 
        dfs(board, r, c - 1); 
        dfs(board, r, c + 1); 
    }
}
