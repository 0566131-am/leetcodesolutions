class Solution {
    public int sumSubarrayMins(int[] arr) {

     int n=arr.length;
     long sumtotal=0;
     long mod = 1_000_000_007;
     for(int i=0;i<n;i++){
        int cumin=arr[i];
     for(int j=i;j<n;j++){
        cumin=Math.min(cumin,arr[j]);
     sumtotal=(sumtotal+cumin)%mod;
     }
     }
     return(int) sumtotal;
        
    }
