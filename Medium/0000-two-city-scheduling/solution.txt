        [0])));
        
        int totalCost = 0;
        int n = costs.length / 2;
        for (int i = 0; i < costs.length; i++) {
            if (i < n) {
                totalCost += costs[i][1]; 
            } else {
                totalCost += costs[i][0]; 
            }
        }
        
        return totalCost;
    }
