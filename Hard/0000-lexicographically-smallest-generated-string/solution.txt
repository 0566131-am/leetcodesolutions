            for (int j = i + m - 1; j >= i; j--) {
                if (oldAns[j] == '?') { // previously filled with 'a', now change to 'b'
                    ans[j] = 'b';
                    ok = true;
                    break;
                }
            }
            if (!ok) {
                return "";
            }
        }

        return new String(ans);
    }
}
