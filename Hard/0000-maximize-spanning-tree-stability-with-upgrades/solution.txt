        for (int[] e : upgradeCandidates) {

            if (usedEdges == n - 1) break;
            if (upgrades >= k) break;

            int u = e[0], v = e[1];

            if (dsu.union(u, v)) {
                usedEdges++;
                upgrades++;
            }
        }

        return usedEdges == n - 1;
    }
}
