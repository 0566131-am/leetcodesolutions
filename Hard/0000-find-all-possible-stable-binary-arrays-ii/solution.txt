                if (i - limit - 1 >= 0) {
                    val0 = (val0 - dp1[i - limit - 1][j] + MOD) % MOD;
                }
                dp0[i][j] = val0;

                long val1 = (dp0[i][j - 1] + dp1[i][j - 1]) % MOD;
                if (j - limit - 1 >= 0) {

                }
                dp1[i][j] = val1;
            }
        }

        return (int)((dp0[zero][one] + dp1[zero][one]) % MOD);
    }
